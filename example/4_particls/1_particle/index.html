var position_vector = new Vector( 50,100);
var velocity_vector = new Vector( 100, -200);
var acceleration_vector = new Vector( 0, 0);

//creating the particle
var myParticle = new Particle();
//creating the force
var myForce = new TestForce();

//setting the particle
myParticle.position = position_vector;
myParticle.velocity = velocity_vector;
myParticle.acceleration = acceleration_vector;
myParticle.mass = 20;

//calculation of the force
myForce.calc_object(myParticle);

//calculation of the particle's position, velocity, acceleration.
myParticle.update(myForce.force)
